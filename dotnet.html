Result Pattern

// Result type for domain operations
public record Result<T>
{
    public T? Value { get; init; }
    public DomainError? Error { get; init; }
    public bool IsSuccess => Error is null;

    public static Result<T> Ok(T value) =>
        new() { Value = value };
    public static Result<T> Fail(DomainError err) =>
        new() { Error = err };
}

// Usage in risk gate
public Result<RiskApproval> CheckOrder(Order o)
{
    if (o.Quantity <= 0)
        return Result<RiskApproval>.Fail(
            new DomainError("Invalid quantity"));

    if (GetExposure(o.Symbol) > _maxExposure)
        return Result<RiskApproval>.Fail(
            new DomainError("Exposure limit"));

    return Result<RiskApproval>.Ok(
        new RiskApproval(o.OrderId));
}

gRPC in .NET

// risk_gate.proto
syntax = "proto3";
package trading;

service RiskGate {
  rpc CheckOrder (OrderRequest) returns (RiskDecision);
  rpc StreamPositions (Empty) returns (stream PositionUpdate);
}

message OrderRequest {
  string order_id = 1;
  string symbol = 2;
  double quantity = 3;
  double price = 4;
  string side = 5;  // BUY or SELL
}

message RiskDecision {
  bool approved = 1;
  string reason = 2;
  double exposure_after = 3;
}

// C# gRPC service implementation
public class RiskGateService(
    IPositionTracker positions,
    IOptionsMonitor<RiskOptions> riskOptions)
    : RiskGate.RiskGateBase
{
    public override async Task<RiskDecision> CheckOrder(
        OrderRequest request, ServerCallContext context)
    {
        var opts = riskOptions.CurrentValue;
        var currentExposure = await positions.GetExposureAsync(request.Symbol);
        var orderValue = (decimal)(request.Quantity * request.Price);

        if (currentExposure + orderValue > opts.ExposureLimitUsd)
            return new RiskDecision { Approved = false, Reason = "Exposure limit" };

        return new RiskDecision { Approved = true, ExposureAfter = (double)(currentExposure + orderValue) };
    }
}

// Client call (<50ms target)
var decision = await _riskClient.CheckOrderAsync(request, deadline: DateTime.UtcNow.AddMilliseconds(50));

Consumer

// Kafka consumer for crypto ticks
public class TickConsumerService(
    IConsumer<string, string> consumer,
    ChannelWriter<Tick> writer,
    ILogger<TickConsumerService> logger)
    : BackgroundService
{
    protected override async Task ExecuteAsync(
        CancellationToken ct)
    {
        consumer.Subscribe("crypto.ticks");

        while (!ct.IsCancellationRequested)
        {
            var result = consumer.Consume(ct);
            var tick = JsonSerializer
                .Deserialize<Tick>(result.Message.Value)!;

            await writer.WriteAsync(tick, ct);

            // Manual commit for at-least-once
            consumer.Commit(result);
        }
    }
}

Background Services & Hosted Services

// Long-running background service for continuous processing
public class FeatureComputeService(
    ChannelReader<Tick> tickReader,
    IFeatureProvider<Tick[], double>[] providers,
    IEventBus eventBus,
    ILogger<FeatureComputeService> logger)
    : BackgroundService
{
    private readonly List<Tick> _window = new(capacity: 200);

    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        logger.LogInformation("Feature compute service started");

        await foreach (var tick in tickReader.ReadAllAsync(ct))
        {
            _window.Add(tick);
            if (_window.Count < 20) continue;  // wait for enough data

            // Compute all features in parallel
            var ticks = _window.ToArray();
            var tasks = providers.Select(p => p.ComputeAsync(ticks, ct));
            var features = await Task.WhenAll(tasks);

            await eventBus.PublishAsync(new FeaturesComputed(tick.Symbol, features), ct);

            // Sliding window: remove oldest
            if (_window.Count > 200)
                _window.RemoveRange(0, _window.Count - 200);
        }
    }
}

// Timed background service for periodic health checks
public class ExchangeHealthCheck(IHttpClientFactory httpFactory) : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        using var timer = new PeriodicTimer(TimeSpan.FromSeconds(30));
        while (await timer.WaitForNextTickAsync(ct))
        {
            var client = httpFactory.CreateClient("exchange");
            var healthy = (await client.GetAsync("/health", ct)).IsSuccessStatusCode;
            ExchangeMetrics.HealthStatus.Set(healthy ? 1 : 0);
        }
    }
}

// Custom middleware for request timing
public class RequestTimingMiddleware(
    RequestDelegate next,
    ILogger<RequestTimingMiddleware> logger)
{
    public async Task InvokeAsync(HttpContext ctx)
    {
        var sw = Stopwatch.StartNew();
        ctx.Response.OnStarting(() =>
        {
            sw.Stop();
            ctx.Response.Headers["X-Response-Time"]
                = $"{sw.ElapsedMilliseconds}ms";
            return Task.CompletedTask;
        });
        await next(ctx);

        logger.LogInformation(
            "{Method} {Path} => {Status} in {Ms}ms",
            ctx.Request.Method, ctx.Request.Path,
            ctx.Response.StatusCode, sw.ElapsedMilliseconds);
    }
}

Best Practices

1. Use records for immutable domain events and value objects
2. Prefer async/await for all I/O; never block with .Result or .Wait()
3. Always pass CancellationToken through the call chain
4. Use IOptions/IOptionsMonitor for configuration, not static values
5. Register services with correct lifetimes (avoid captive dependencies)
6. Use Channels for producer/consumer pipelines over shared collections
7. Seal classes that are not designed for inheritance
8. Use Span<T> and ArrayPool on hot paths to minimize GC pressure
9. Write parameterized tests with [Theory] for boundary conditions
10. Use Result pattern over exceptions for expected domain failures
