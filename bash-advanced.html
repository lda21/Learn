Strict Mode & Defensive Scripting

Every production script should start with this. It turns silent failures into loud, immediate errors:

#!/usr/bin/env bash
set -euo pipefail
IFS=$'\n\t'

# set -e   : Exit immediately on any command failure
# set -u   : Treat unset variables as errors (catches typos)
# set -o pipefail : Pipe fails if ANY command in the pipeline fails
#                   (not just the last one)
# IFS      : Safer word splitting (no spaces in filenames breaking things)

Without Strict Mode

# Silent disaster
rm -rf "$DEPLOY_DIR"/$TYPO_VAR
# $TYPO_VAR is empty, so this
# becomes: rm -rf "$DEPLOY_DIR"/
# No error, just destruction

With Strict Mode

set -euo pipefail
rm -rf "$DEPLOY_DIR"/$TYPO_VAR
# bash: TYPO_VAR: unbound variable
# Script exits immediately
# Nothing gets deleted

Error Handling Patterns

#!/usr/bin/env bash
set -euo pipefail

# Cleanup function runs on ANY exit (success, error, or interrupt)
cleanup() {
  local exit_code=$?
  echo "Cleaning up... (exit code: $exit_code)"
  rm -f "$TMPFILE"
  [[ -n "${PID:-}" ]] && kill "$PID" 2>/dev/null || true
  exit "$exit_code"
}
trap cleanup EXIT        # Always runs on script exit
trap 'echo "Interrupted!"; exit 130' INT TERM  # Ctrl+C / kill

TMPFILE=$(mktemp)

# Allow specific commands to fail without killing the script
grep -q "pattern" file.txt || true          # Ignore grep failure
result=$(some_command 2>/dev/null) || result="default"

# Custom error handler with line numbers
err_handler() {
  echo "ERROR: line $1, command: $2, exit code: $3" >&2
}
trap 'err_handler ${LINENO} "${BASH_COMMAND}" $?' ERR

trap EXIT is the single most important pattern — it guarantees cleanup regardless of how the script ends.

Functions & Return Values

# Local variables prevent leaking into global scope
process_file() {
  local filename="$1"
  local -r MAX_SIZE=1048576    # -r = readonly
  local size
  size=$(stat -f%z "$filename" 2>/dev/null || stat -c%s "$filename")

  if (( size > MAX_SIZE )); then
    return 1                   # return = exit code only (0-255)
  fi
  echo "$size"                 # Use echo/printf for actual return data
}

# Capture function output via subshell
file_size=$(process_file "/var/log/syslog") || echo "File too large"

# Nameref variables (bash 4.3+) — pass by reference, no subshell
populate_array() {
  local -n result_ref="$1"    # -n creates a nameref
  result_ref=("alpha" "bravo" "charlie")
}
declare -a my_data
populate_array my_data
echo "${my_data[1]}"          # "bravo"

# Function libraries — source reusable code
source /usr/local/lib/logging.sh   # Import log_info, log_error, etc.
source ./lib/validation.sh          # Import validate_email, validate_ip

Arrays & Associative Arrays

# Indexed arrays
servers=("web-01" "web-02" "db-01" "cache-01")
echo "${servers[0]}"              # web-01
echo "${servers[@]}"              # All elements
echo "${#servers[@]}"             # Length: 4
echo "${servers[@]:1:2}"          # Slice: web-02 db-01

# Append and delete
servers+=("worker-01")
unset 'servers[2]'                # Remove db-01 (leaves gap)
servers=("${servers[@]}")         # Re-index to close gap

# Associative arrays (bash 4.0+)
declare -A config
config[db_host]="postgres.internal"
config[db_port]="5432"
config[db_name]="production"

# Iterate keys and values
for key in "${!config[@]}"; do
  echo "$key = ${config[$key]}"
done

# Pass array to function (by name reference)
process_servers() {
  local -n arr="$1"
  for server in "${arr[@]}"; do
    echo "Deploying to: $server"
  done
}
process_servers servers

String Manipulation

Parameter expansion is faster than calling external tools (sed, awk, tr) for simple operations:

filepath="/var/log/nginx/access.log.gz"

# Strip patterns
echo "${filepath#*/}"           # var/log/nginx/access.log.gz  (shortest from start)
echo "${filepath##*/}"          # access.log.gz                (longest from start)
echo "${filepath%.*}"           # /var/log/nginx/access.log    (shortest from end)
echo "${filepath%%.*}"          # /var/log/nginx/access        (longest from end)

# Substitution
version="v1.2.3-beta"
echo "${version/beta/rc1}"      # v1.2.3-rc1     (first match)
echo "${version//./_ }"         # v1_2_3-beta     (all matches — note: / / = global)

# Substring
echo "${version:1:5}"           # 1.2.3           (offset:length)
echo "${version: -4}"           # beta            (last 4 chars, space before -)

# Case conversion (bash 4.0+)
name="hello world"
echo "${name^^}"                # HELLO WORLD     (uppercase all)
echo "${name^}"                 # Hello world     (uppercase first)
upper="LOUD"
echo "${upper,,}"               # loud            (lowercase all)

# Default values
echo "${DB_HOST:-localhost}"    # Use localhost if DB_HOST is unset or empty
echo "${DB_HOST:=localhost}"    # Same, but also assigns the default
: "${REQUIRED_VAR:?'REQUIRED_VAR must be set'}"  # Exit with error if unset

read -r first rest <<< "hello world"

Subshells & Process Management

# () = subshell — isolated environment, cannot affect parent
(cd /tmp && rm -f *.log)     # Parent's cwd is unchanged
val=1; (val=2; echo "$val"); echo "$val"  # Prints: 2, then 1

# {} = group — runs in CURRENT shell, shares environment
{ cd /tmp; ls; }              # Parent's cwd IS changed

# Background processes
long_task_1 &                 # Run in background, get PID
pid1=$!
long_task_2 &
pid2=$!

# Wait for specific processes
wait "$pid1" "$pid2"          # Wait for both to finish
echo "All tasks complete"

# Wait for any single process to finish
wait -n "$pid1" "$pid2"       # Returns when first one completes

# Disown — detach from terminal
nohup ./server.sh > /var/log/server.log 2>&1 &
disown                        # Process survives terminal close

# Process group management
./parent.sh &
pgid=$!
kill -- -"$pgid"              # Kill entire process group (note the -)

Use () when you need isolation. Use {} when you need grouping without overhead.

#!/usr/bin/env bash
set -euo pipefail
SHUTDOWN=0

# Graceful shutdown pattern for long-running services
graceful_shutdown() {
  echo "Received shutdown signal, finishing current work..."
  SHUTDOWN=1
  # Don't exit here — let the main loop finish its current iteration
}
trap graceful_shutdown SIGTERM SIGINT

# Main processing loop
while (( SHUTDOWN == 0 )); do
  process_next_job            # Finish current job before exiting
  sleep 1
done
echo "Clean shutdown complete"

# Config reload on SIGHUP (kill -HUP <pid>)
trap 'source /etc/myapp.conf; echo "Config reloaded"' SIGHUP

File Descriptors & Redirection

# Standard FDs: 0=stdin, 1=stdout, 2=stderr
# Custom FDs: 3-9 are available

# Open FD 3 for writing to a log file
exec 3>/var/log/app-debug.log
echo "Debug message" >&3       # Write to FD 3
echo "Normal output"            # Goes to stdout
exec 3>&-                       # Close FD 3

# Open FD 4 for reading
exec 4</etc/config.ini
while IFS= read -r line <&4; do
  echo "Config: $line"
done
exec 4<&-                       # Close FD 4

# Swap stdout and stderr
command 3>&1 1>&2 2>&3 3>&-

# Capture stderr separately while displaying stdout
{ error_output=$(command 2>&1 1>&3); } 3>&1
echo "Errors were: $error_output"

# Named pipes (FIFOs) — inter-process communication
mkfifo /tmp/mypipe
producer_command > /tmp/mypipe &
consumer_command < /tmp/mypipe
rm /tmp/mypipe

# Tee pattern — log everything while still displaying
exec > >(tee -a /var/log/script.log) 2>&1
echo "This goes to both terminal and log file"

# Background jobs with controlled concurrency
MAX_JOBS=4
pids=()

for server in "${servers[@]}"; do
  deploy_to "$server" &
  pids+=($!)

  # Throttle: if we hit max, wait for one to finish
  if (( ${#pids[@]} >= MAX_JOBS )); then
    wait -n                               # Wait for ANY one to finish
    # Clean up finished PIDs
    new_pids=()
    for pid in "${pids[@]}"; do
      kill -0 "$pid" 2>/dev/null && new_pids+=("$pid")
    done
    pids=("${new_pids[@]}")
  fi
done
wait                                      # Wait for remaining jobs

# Collect exit codes from parallel jobs
declare -A results
for server in "${servers[@]}"; do
  check_health "$server" &
  results["$server"]=$!
done
for server in "${!results[@]}"; do
  wait "${results[$server]}" && echo "$server: OK" || echo "$server: FAIL"
done

Argument Parsing

#!/usr/bin/env bash
set -euo pipefail

VERSION="1.0.0"
VERBOSE=false
OUTPUT=""
ENVIRONMENT="staging"

usage() {
  cat <<EOF
Usage: $(basename "$0") [OPTIONS] <input-file>

Options:
  -e, --env ENV        Target environment (default: staging)
  -o, --output FILE    Output file path
  -v, --verbose        Enable verbose output
  -h, --help           Show this help message
  --version            Show version
  --                   End of options
EOF
  exit "${1:-0}"
}

# Manual parsing — supports long options (getopts doesn't)
while [[ $# -gt 0 ]]; do
  case "$1" in
    -e|--env)      ENVIRONMENT="$2"; shift 2 ;;
    -o|--output)   OUTPUT="$2"; shift 2 ;;
    -v|--verbose)  VERBOSE=true; shift ;;
    -h|--help)     usage 0 ;;
    --version)     echo "$VERSION"; exit 0 ;;
    --)            shift; break ;;       # End of options
    -*)            echo "Unknown option: $1" >&2; usage 1 ;;
    *)             break ;;              # First non-option argument
  esac
done

# Validate required positional argument
INPUT_FILE="${1:?'Error: input file required. Use --help for usage.'}"
[[ -f "$INPUT_FILE" ]] || { echo "File not found: $INPUT_FILE" >&2; exit 1; }

Logging & Output

#!/usr/bin/env bash
# Color codes (only when outputting to terminal)
if [[ -t 1 ]]; then
  RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[1;33m'
  BLUE='\033[0;34m'; BOLD='\033[1m'; NC='\033[0m'
else
  RED=''; GREEN=''; YELLOW=''; BLUE=''; BOLD=''; NC=''
fi

# Log levels — always write logs to stderr (keep stdout for data)
log_info()  { echo -e "${GREEN}[INFO]${NC}  $(date +%H:%M:%S) $*" >&2; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC}  $(date +%H:%M:%S) $*" >&2; }
log_error() { echo -e "${RED}[ERROR]${NC} $(date +%H:%M:%S) $*" >&2; }
log_debug() { [[ "${DEBUG:-}" == "true" ]] && echo -e "${BLUE}[DEBUG]${NC} $(date +%H:%M:%S) $*" >&2; }

# Structured logging for machine parsing
log_json() {
  printf '{"ts":"%s","level":"%s","msg":"%s"}\n' \
    "$(date -u +%Y-%m-%dT%H:%M:%SZ)" "$1" "$2" >&2
}

# Progress indicator for long operations
spinner() {
  local pid=$1 msg="${2:-Processing}"
  local chars='|/-\'
  while kill -0 "$pid" 2>/dev/null; do
    for (( i=0; i<${#chars}; i++ )); do
      printf '\r%s %s ' "$msg" "${chars:$i:1}" >&2
      sleep 0.1
    done
  done
  printf '\r%s done.\n' "$msg" >&2
}

heavy_computation &
spinner $! "Compiling"

Configuration Management

#!/usr/bin/env bash
set -euo pipefail

# Defaults — every config var has a fallback
DB_HOST="${DB_HOST:-localhost}"
DB_PORT="${DB_PORT:-5432}"
LOG_LEVEL="${LOG_LEVEL:-info}"
RETRY_COUNT="${RETRY_COUNT:-3}"

# Source config file safely (don't fail if missing)
CONFIG_FILE="${CONFIG_FILE:-/etc/myapp/config.sh}"
if [[ -f "$CONFIG_FILE" ]]; then
  # shellcheck source=/dev/null
  source "$CONFIG_FILE"
fi

# Parse .env file (skip comments and empty lines)
load_env() {
  local env_file="${1:-.env}"
  [[ -f "$env_file" ]] || return 0
  while IFS= read -r line; do
    # Skip comments and empty lines
    [[ -z "$line" || "$line" == \#* ]] && continue
    # Only export if not already set (don't override environment)
    local key="${line%%=*}"
    if [[ -z "${!key:-}" ]]; then
      export "$line"
    fi
  done < "$env_file"
}
load_env

# Validate required configuration
validate_config() {
  local missing=()
  [[ -z "${DATABASE_URL:-}" ]]  && missing+=("DATABASE_URL")
  [[ -z "${API_KEY:-}" ]]       && missing+=("API_KEY")
  [[ -z "${S3_BUCKET:-}" ]]     && missing+=("S3_BUCKET")

  if (( ${#missing[@]} > 0 )); then
    echo "Missing required config: ${missing[*]}" >&2
    exit 1
  fi
}
validate_config

# Retry logic with exponential backoff
retry() {
  local max_attempts="${1}"; shift
  local delay=1
  for (( attempt=1; attempt <= max_attempts; attempt++ )); do
    if "$@"; then return 0; fi
    echo "Attempt $attempt/$max_attempts failed. Retrying in ${delay}s..." >&2
    sleep "$delay"
    delay=$((delay * 2))
  done
  echo "All $max_attempts attempts failed" >&2
  return 1
}

Docker & Container Scripting

#!/usr/bin/env bash
set -euo pipefail

APP="myapp"
TAG="${GIT_SHA:-$(git rev-parse --short HEAD)}"
REGISTRY="123456789.dkr.ecr.us-east-1.amazonaws.com"
IMAGE="${REGISTRY}/${APP}:${TAG}"

# Build and tag
docker build -t "${IMAGE}" -t "${REGISTRY}/${APP}:latest" .

# Health check before deploying new version
wait_healthy() {
  local container="$1" timeout="${2:-60}"
  for (( i=0; i<timeout; i++ )); do
    status=$(docker inspect --format='{{.State.Health.Status}}' "$container" 2>/dev/null || echo "missing")
    [[ "$status" == "healthy" ]] && return 0
    sleep 1
  done
  echo "Container $container not healthy after ${timeout}s" >&2
  docker logs "$container" --tail 20 >&2
  return 1
}

# Docker Compose automation
docker compose -f docker-compose.prod.yml pull
docker compose -f docker-compose.prod.yml up -d --remove-orphans
wait_healthy "${APP}-web-1" 90

# Container log parsing — find errors in last hour
docker logs "${APP}-web-1" --since 1h 2>&1 | grep -i "error\|fatal\|panic"

# Cleanup old images (keep last 5)
docker images "${REGISTRY}/${APP}" --format '{{.Tag}}' | \
  sort -rV | tail -n +6 | \
  xargs -I {} docker rmi "${REGISTRY}/${APP}:{}" 2>/dev/null || true

fi

echo "::endgroup::"
echo "::notice::Deployment to ${ENVIRONMENT} complete"

Performance & Debugging

# Trace execution with set -x and custom PS4
export PS4='+${BASH_SOURCE}:${LINENO}:${FUNCNAME[0]:+${FUNCNAME[0]}(): }'
set -x                                 # Shows every command before execution
# Output: +script.sh:42:deploy(): curl -s https://...

# Send trace to a file instead of stderr
exec {BASH_XTRACEFD}>/tmp/trace.log
set -x
# ... your script ...
set +x
exec {BASH_XTRACEFD}>&-

# Time commands
time expensive_operation                # Real, user, sys time
SECONDS=0                              # Built-in timer
do_work
echo "Took ${SECONDS} seconds"

# strace — trace system calls (debug hanging scripts)
strace -f -e trace=network bash script.sh   # Network calls only
strace -e trace=file bash script.sh         # File operations only

Avoid Useless cat

cat file | grep x → grep x file. Saves a process fork on every call.

Avoid Subshell Loops

cmd | while read runs in a subshell. Use while read < <(cmd) instead to keep variables.

Use Built-ins

${var//x/y} is faster than echo "$var" | sed. Parameter expansion avoids forking external processes.

Batch Operations

find . -exec cmd {} + batches arguments. find . -exec cmd {} \; forks once per file.

Real-World Patterns

# === Deployment Script Skeleton ===
#!/usr/bin/env bash
set -euo pipefail
trap 'echo "Deploy failed at line $LINENO" >&2; rollback' ERR
trap cleanup EXIT

DEPLOY_DIR="/opt/app"
BACKUP_DIR="/opt/app-backup-$(date +%s)"

rollback() { [[ -d "$BACKUP_DIR" ]] && mv "$BACKUP_DIR" "$DEPLOY_DIR"; }
cleanup()  { rm -rf "${TMPDIR:-}"; }

cp -a "$DEPLOY_DIR" "$BACKUP_DIR"       # Backup current
rsync -a --delete dist/ "$DEPLOY_DIR/"   # Deploy new
systemctl restart myapp                  # Restart service
retry 5 curl -sf http://localhost:8080/health  # Verify
rm -rf "$BACKUP_DIR"                     # Success — remove backup

# === Database Backup with Rotation ===
BACKUP_FILE="db-$(date +%Y%m%d-%H%M%S).sql.gz"
pg_dump "$DATABASE_URL" | gzip > "/backups/$BACKUP_FILE"
aws s3 cp "/backups/$BACKUP_FILE" "s3://${S3_BUCKET}/db/"
find /backups -name "db-*.sql.gz" -mtime +7 -delete   # Keep 7 days local

# === Health Check Script ===
check_endpoint() {
  local url="$1" name="$2"
  local code
  code=$(curl -so /dev/null -w '%{http_code}' --max-time 5 "$url") || code="000"
  if [[ "$code" != "200" ]]; then
    echo "CRITICAL: $name returned $code" >&2
    return 1
  fi
}
check_endpoint "http://localhost:8080/health" "API"
check_endpoint "http://localhost:3000" "Frontend"
