What is Linux?

  • An open-source Unix-like operating system kernel created by Linus Torvalds (1991)
  • Powers 90%+ of servers, all Android devices, and most cloud infrastructure
  • Distributions (distros): Ubuntu, CentOS, Debian, Arch, RHEL, Amazon Linux
  • The backbone of DevOps, cloud computing, and containerization
  • Free, customizable, and supported by a massive community

File System Hierarchy

/               # Root of everything
├── bin/        # Essential user binaries (ls, cp, mv)
├── etc/        # System configuration files
├── home/       # User home directories
├── var/        # Variable data (logs, databases)
│   └── log/    # System and app logs
├── tmp/        # Temporary files
├── usr/        # User programs and data
│   ├── bin/    # Most user commands
│   └── lib/    # Libraries
├── opt/        # Optional/third-party software
├── dev/        # Device files
├── proc/       # Process and kernel info (virtual)
└── mnt/        # Mount points for filesystems

Navigation & File Operations

# Navigation
pwd                         # Print working directory
cd /var/log                 # Change directory
cd ~                        # Go to home directory
cd -                        # Go to previous directory

# Listing
ls -la                      # List all files with details
ls -lh                      # Human-readable file sizes
tree -L 2                   # Tree view, 2 levels deep

# File operations
cp -r source/ dest/         # Copy recursively
mv old.txt new.txt          # Move / rename
rm -rf directory/           # Remove recursively (careful!)
mkdir -p a/b/c              # Create nested directories
touch file.txt              # Create empty file
ln -s target link           # Create symbolic link

File Permissions

# Permission format: rwxrwxrwx (owner|group|others)
# r=read(4) w=write(2) x=execute(1)

-rw-r--r--  1 user group  4096 Mar 15 10:30 file.txt
drwxr-xr-x  2 user group  4096 Mar 15 10:30 directory/

# Change permissions
chmod 755 script.sh         # rwxr-xr-x
chmod u+x script.sh         # Add execute for owner
chmod -R 644 dir/           # Recursive

# Change ownership
chown user:group file.txt
chown -R user:group dir/

# Special: setuid, setgid, sticky bit
chmod 4755 program          # setuid
chmod 1777 /tmp             # sticky bit

Text Processing

# View files
cat file.txt                # Print entire file
less file.txt               # Scrollable viewer
head -n 20 file.txt         # First 20 lines
tail -f /var/log/syslog     # Follow log in real-time

# Search
grep -r "error" /var/log/   # Recursive search
grep -i "pattern" file      # Case-insensitive
grep -c "pattern" file      # Count matches

# Transform
sed 's/old/new/g' file      # Find and replace
awk '{print $1, $3}' file   # Print columns 1 and 3
sort file.txt               # Sort lines
uniq -c                     # Count unique lines
wc -l file.txt              # Count lines
cut -d',' -f1,3 data.csv    # Extract CSV columns

Pipes & Redirection

# Pipes: connect output of one command to input of another
cat access.log | grep "500" | wc -l

# Count unique IPs from a log
awk '{print $1}' access.log | sort | uniq -c | sort -rn | head -10

# Redirection
echo "hello" > file.txt     # Write (overwrite)
echo "world" >> file.txt    # Append
command 2> errors.log       # Redirect stderr
command > out.log 2>&1      # Redirect both stdout & stderr
command < input.txt         # Redirect stdin

# /dev/null - discard output
command > /dev/null 2>&1

# Here document
cat << EOF > config.yml
name: myapp
port: 8080
EOF

Process Management

# View processes
ps aux                      # All processes
ps aux | grep nginx         # Find specific process
top                         # Interactive process viewer
htop                        # Better interactive viewer

# Process control
command &                   # Run in background
jobs                        # List background jobs
fg %1                       # Bring job 1 to foreground
nohup command &             # Survive terminal close

# Kill processes
kill PID                    # Graceful terminate (SIGTERM)
kill -9 PID                 # Force kill (SIGKILL)
killall nginx               # Kill by name
pkill -f "pattern"          # Kill by pattern

# System services (systemd)
systemctl start nginx
systemctl enable nginx      # Start on boot
systemctl status nginx
journalctl -u nginx -f      # View service logs

Networking

# Network info
ip addr show                # Show IP addresses
ip route                    # Show routing table
ss -tlnp                    # Show listening ports
hostname -I                 # Quick IP lookup

# Connectivity
ping google.com             # Test connectivity
curl -I https://example.com # HTTP headers
wget https://example.com/f  # Download file
traceroute google.com       # Trace network path

# DNS
dig example.com             # DNS lookup
nslookup example.com        # DNS query
cat /etc/resolv.conf        # DNS config

# Firewall (iptables / firewalld)
ufw allow 22/tcp            # Allow SSH (Ubuntu)
ufw enable                  # Enable firewall
firewall-cmd --add-port=80/tcp  # (CentOS/RHEL)

SSH & Remote Access

# Connect
ssh user@server.com
ssh -i key.pem ec2-user@ip  # With private key
ssh -p 2222 user@host       # Custom port

# Key management
ssh-keygen -t ed25519       # Generate key pair
ssh-copy-id user@server     # Copy public key to server
cat ~/.ssh/authorized_keys  # Authorized keys on server

# SSH config (~/.ssh/config)
Host myserver
  HostName 192.168.1.100
  User admin
  IdentityFile ~/.ssh/mykey
  Port 22

# Then just: ssh myserver

# File transfer
scp file.txt user@server:/path/
scp -r dir/ user@server:/path/
rsync -avz dir/ user@server:/path/  # Better for large transfers

# Tunneling
ssh -L 8080:localhost:3000 user@server  # Local port forward

Shell Scripting Basics

#!/bin/bash

# Variables
NAME="World"
echo "Hello, $NAME!"

# Conditionals
if [ -f "/etc/nginx/nginx.conf" ]; then
    echo "Nginx config exists"
elif [ -d "/etc/apache2" ]; then
    echo "Apache installed"
else
    echo "No web server found"
fi

# Loops
for server in web1 web2 web3; do
    ssh "$server" "systemctl restart nginx"
done

# Functions
deploy() {
    local app=$1
    echo "Deploying $app..."
    rsync -avz ./build/ "prod:/var/www/$app/"
}
deploy "myapp"

# Exit codes: 0 = success, non-zero = failure

Cron & Scheduled Tasks

# Cron format: minute hour day month weekday command
# Edit crontab
crontab -e                  # Edit current user's crontab
crontab -l                  # List cron jobs

# Examples
# Every 5 minutes
*/5 * * * * /scripts/check-health.sh

# Daily at 2:30 AM
30 2 * * * /scripts/backup.sh >> /var/log/backup.log 2>&1

# Every Monday at 9 AM
0 9 * * 1 /scripts/weekly-report.sh

# First day of every month
0 0 1 * * /scripts/monthly-cleanup.sh

# System cron directories
/etc/cron.daily/            # Runs daily
/etc/cron.weekly/           # Runs weekly
/etc/cron.monthly/          # Runs monthly

Security Essentials

1 Disable root SSH login — use PermitRootLogin no
2 Use SSH keys instead of passwords
3 Keep system updated — apt upgrade regularly
4 Configure firewall — only open needed ports
5 Use fail2ban to block brute-force attempts
6 Follow least privilege — avoid running as root
7 Monitor logs with journalctl and log aggregation
8 Use SELinux / AppArmor for mandatory access control
