What is Vercel?

  • A frontend cloud platform built by the creators of Next.js
  • Deploy any frontend framework: Next.js, React, Vue, Svelte, Astro, and more
  • Global Edge Network with automatic SSL, CDN, and instant rollbacks
  • Built-in Serverless Functions and Edge Functions for backend logic
  • Automatic preview deployments for every Git push — no CI/CD config needed

Serverless Functions

// api/hello.ts - Node.js serverless function
import type { VercelRequest, VercelResponse } from '@vercel/node';

export default function handler(req: VercelRequest, res: VercelResponse) {
  const { name = 'World' } = req.query;
  res.status(200).json({ message: `Hello ${name}!` });
}
// Accessible at: /api/hello?name=Vercel
      
# api/hello.py - Python serverless function
from http.server import BaseHTTPRequestHandler
import json

class handler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.send_header('Content-Type', 'application/json')
        self.end_headers()
        self.wfile.write(json.dumps({"message": "Hello from Python!"}).encode())
      
// vercel.json - Function configuration
{
  "functions": {
    "api/heavy-task.ts": {
      "memory": 3008,         // MB (128-3008)
      "maxDuration": 60,      // seconds (Hobby: 10s, Pro: 60s, Enterprise: 900s)
      "regions": ["iad1"]     // Pin to specific region
    }
  }
}

Edge Functions

// middleware.ts - Edge Middleware (runs before every request)
import { NextRequest, NextResponse } from 'next/server';

export function middleware(request: NextRequest) {
  // Geo-aware routing
  const country = request.geo?.country || 'US';

  // A/B testing at the edge
  const bucket = request.cookies.get('ab-bucket')?.value
    || (Math.random() > 0.5 ? 'a' : 'b');

  const response = NextResponse.next();
  response.cookies.set('ab-bucket', bucket);

  // Redirect based on country
  if (country === 'DE') {
    return NextResponse.rewrite(new URL('/de' + request.nextUrl.pathname, request.url));
  }
  return response;
}

export const config = {
  matcher: ['/((?!api|_next/static|favicon.ico).*)']
};
      
// app/api/edge-route/route.ts - Edge API Route
export const runtime = 'edge';  // Use edge runtime instead of Node.js

export async function GET(request: Request) {
  return new Response(JSON.stringify({ edge: true, region: process.env.VERCEL_REGION }), {
    headers: { 'Content-Type': 'application/json' }
  });
}
// Edge: ~0ms cold start, limited APIs (no fs, no native modules)
// Node: ~250ms cold start, full Node.js APIs

Vercel & Next.js

SSR (Server-Side Rendering)

Pages rendered on every request via serverless functions. Full dynamic content.

export const dynamic = 'force-dynamic';

SSG (Static Site Generation)

Pages pre-rendered at build time. Served from CDN. Fastest possible response.

generateStaticParams()

ISR (Incremental Static Regeneration)

Static pages that revalidate in the background. Best of both worlds.

export const revalidate = 60;

Server Components

React components that run on the server. Zero client-side JS. Stream HTML to the browser.

// default in App Router 
// app/products/[id]/page.tsx - ISR with App Router
export const revalidate = 3600; // Revalidate every hour

export async function generateStaticParams() {
  const products = await fetch('https://api.example.com/products').then(r => r.json());
  return products.map((p: any) => ({ id: p.id.toString() }));
}

export default async function ProductPage({ params }: { params: { id: string } }) {
  const product = await fetch(`https://api.example.com/products/${params.id}`).then(r => r.json());
  return <div>{product.name} - ${product.price}</div>;
}

Best Practices

1 Use ISR over SSR where possible — static with revalidation is faster and cheaper
2 Set Cache-Control headers — use s-maxage and stale-while-revalidate
3 Use Edge Functions for auth, redirects, and A/B tests — near-zero cold starts
4 Pin serverless functions to one region near your database
5 Use vercel env pull to keep local env in sync — never hardcode secrets
6 Monitor Web Vitals in production — track LCP, CLS, INP regressions
7 Use Edge Config for feature flags — 15ms reads globally, no cold starts
8 Set spending limits — prevent unexpected bills from traffic spikes
9 Use ignoreCommand in vercel.json — skip unnecessary builds on docs-only changes
10 Enable Deployment Protection for preview URLs — prevent unauthorized access
