What is Playwright?

Overview

  • E2E testing framework by Microsoft
  • Chromium, Firefox, WebKit — one API
  • Auto-waiting for elements & actions
  • Runs on Windows, macOS, Linux, CI
  • TypeScript, JavaScript, Python, Java, C#

Why Playwright?

  • Auto-wait — no manual sleep/wait
  • Test isolation — browser contexts
  • Parallelism — built-in workers
  • Codegen — record tests visually
  • Trace viewer — time-travel debugging

Installation & Setup

# Create a new Playwright project
npm init playwright@latest

# Project structure created:
# playwright.config.ts   - configuration
# tests/                  - test files
# tests/example.spec.ts  - sample test
# .github/workflows/     - CI config (optional)

// playwright.config.ts
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './tests',
  fullyParallel: true,
  retries: process.env.CI ? 2 : 0,
  workers: process.env.CI ? 1 : undefined,
  reporter: 'html',
  use: {
    baseURL: 'http://localhost:3000',
    trace: 'on-first-retry',
  },
  projects: [
    { name: 'chromium', use: { ...devices['Desktop Chrome'] } },
    { name: 'firefox', use: { ...devices['Desktop Firefox'] } },
    { name: 'webkit', use: { ...devices['Desktop Safari'] } },
  ],
});

Writing Your First Test

// tests/example.spec.ts
import { test, expect } from '@playwright/test';

test('homepage has title and links', async ({ page }) => {
  await page.goto('https://example.com');

  // Expect the title to contain a substring
  await expect(page).toHaveTitle(/Example/);

  // Click the "More information" link
  await page.getByRole('link', { name: 'More information' }).click();

  // Expect the URL to change
  await expect(page).toHaveURL(/iana.org/);
});

test('user can log in', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('user@test.com');
  await page.getByLabel('Password').fill('secret123');
  await page.getByRole('button', { name: 'Sign in' }).click();
  await expect(page.getByText('Welcome back')).toBeVisible();
});

# Run tests
npx playwright test              # all tests
npx playwright test login.spec   # specific file
npx playwright test --headed     # see the browser

Locators

Locators are Playwright's primary way to find elements. They auto-wait and auto-retry.

LocatorExampleBest For
getByRolepage.getByRole('button', { name: 'Submit' })Accessible elements
getByTextpage.getByText('Welcome')Visible text
getByLabelpage.getByLabel('Email')Form inputs
getByTestIdpage.getByTestId('submit-btn')data-testid attrs
getByPlaceholderpage.getByPlaceholder('Search...')Placeholder text
locatorpage.locator('.card >> h3')CSS/chaining

Priority: getByRole > getByLabel > getByText > getByTestId > CSS selectors

Assertions

// Auto-retrying assertions (recommended)
await expect(locator).toBeVisible();
await expect(locator).toBeHidden();
await expect(locator).toHaveText('Hello');
await expect(locator).toContainText('Hello');
await expect(locator).toHaveAttribute('href', '/about');
await expect(locator).toHaveClass(/active/);
await expect(locator).toHaveCount(3);
await expect(locator).toBeEnabled();
await expect(locator).toBeChecked();

// Page assertions
await expect(page).toHaveTitle(/Dashboard/);
await expect(page).toHaveURL('/dashboard');

// Non-retrying (snapshot)
expect(await page.title()).toBe('Dashboard');

// Negative assertions
await expect(locator).not.toBeVisible();

// Soft assertions (don't stop the test)
await expect.soft(locator).toHaveText('Hello');

Page Object Model

// pages/login.page.ts
import { Page, Locator, expect } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly email: Locator;
  readonly password: Locator;
  readonly submitBtn: Locator;

  constructor(page: Page) {
    this.page = page;
    this.email = page.getByLabel('Email');
    this.password = page.getByLabel('Password');
    this.submitBtn = page.getByRole('button', { name: 'Sign in' });
  }

  async goto() { await this.page.goto('/login'); }

  async login(email: string, password: string) {
    await this.email.fill(email);
    await this.password.fill(password);
    await this.submitBtn.click();
  }
}

// tests/login.spec.ts
test('successful login', async ({ page }) => {
  const loginPage = new LoginPage(page);
  await loginPage.goto();
  await loginPage.login('user@test.com', 'secret');
  await expect(page).toHaveURL('/dashboard');
});

Auto-Waiting & Retries

Auto-Wait (Built-in)

  • Every action waits for element to be actionable
  • Visible, stable, enabled, not obscured
  • No manual waitFor needed
  • Default timeout: 30 seconds
  • Assertions auto-retry until passing

Test Retries

// playwright.config.ts
export default defineConfig({
  retries: process.env.CI ? 2 : 0,
  // Retry failed tests up to 2x in CI
});

// Per-test retry
test('flaky test', async ({ page }) => {
  test.info().annotations.push({
    type: 'retries', description: '3'
  });
});

Custom Timeouts

await expect(locator).toBeVisible({ timeout: 10000 });  // 10s for this assertion
test.setTimeout(120000);                                  // 2 min for this test

Browser Contexts & Fixtures

Every test runs in an isolated browser context — like a fresh incognito window.

// Custom fixture: auto-login for every test
import { test as base, expect } from '@playwright/test';

type MyFixtures = { authedPage: Page };

export const test = base.extend<MyFixtures>({
  authedPage: async ({ browser }, use) => {
    const context = await browser.newContext({
      storageState: 'auth.json',  // saved cookies/localStorage
    });
    const page = await context.newPage();
    await use(page);
    await context.close();
  },
});

// Use in tests
test('dashboard loads', async ({ authedPage }) => {
  await authedPage.goto('/dashboard');
  await expect(authedPage.getByText('Welcome')).toBeVisible();
});

API Testing

import { test, expect } from '@playwright/test';

test('GET /api/users', async ({ request }) => {
  const response = await request.get('/api/users');
  expect(response.status()).toBe(200);

  const users = await response.json();
  expect(users.length).toBeGreaterThan(0);
  expect(users[0]).toHaveProperty('email');
});

test('POST /api/users', async ({ request }) => {
  const response = await request.post('/api/users', {
    data: { name: 'Alice', email: 'alice@test.com' },
  });
  expect(response.status()).toBe(201);
  expect(await response.json()).toMatchObject({
    name: 'Alice',
    email: 'alice@test.com',
  });
});

test('authenticated API call', async ({ request }) => {
  const response = await request.get('/api/profile', {
    headers: { Authorization: 'Bearer token123' },
  });
  expect(response.ok()).toBeTruthy();
});

Authentication Patterns

// global-setup.ts — runs once before all tests
import { chromium } from '@playwright/test';

async function globalSetup() {
  const browser = await chromium.launch();
  const page = await browser.newPage();

  await page.goto('http://localhost:3000/login');
  await page.getByLabel('Email').fill('admin@test.com');
  await page.getByLabel('Password').fill('admin123');
  await page.getByRole('button', { name: 'Sign in' }).click();
  await page.waitForURL('/dashboard');

  // Save authentication state to file
  await page.context().storageState({ path: 'auth.json' });
  await browser.close();
}

export default globalSetup;

// playwright.config.ts
export default defineConfig({
  globalSetup: require.resolve('./global-setup'),
  use: {
    storageState: 'auth.json',  // all tests start logged in
  },
});

Network Interception

// Mock API responses
await page.route('**/api/users', async (route) => {
  await route.fulfill({
    status: 200,
    contentType: 'application/json',
    body: JSON.stringify([{ id: 1, name: 'Mock User' }]),
  });
});

// Abort requests (block images, analytics)
await page.route('**/*.{png,jpg}', route => route.abort());
await page.route('**/analytics/**', route => route.abort());

// Modify response
await page.route('**/api/config', async (route) => {
  const response = await route.fetch();
  const json = await response.json();
  json.featureFlag = true;
  await route.fulfill({ response, body: JSON.stringify(json) });
});

// Record network as HAR file
const context = await browser.newContext({
  recordHar: { path: 'network.har' }
});
// ... run tests ...
await context.close(); // HAR saved

# View traces after test run
npx playwright show-trace \
  test-results/trace.zip

Parallel Execution & Sharding

// playwright.config.ts
export default defineConfig({
  fullyParallel: true,  // run tests within files in parallel
  workers: 4,           // number of parallel workers

  // Or percentage of CPU cores
  workers: process.env.CI ? 1 : '50%',
});

# Run with specific workers
npx playwright test --workers=4

# Shard tests across CI machines
# Machine 1:
npx playwright test --shard=1/3
# Machine 2:
npx playwright test --shard=2/3
# Machine 3:
npx playwright test --shard=3/3

Serial Tests

// Force sequential execution
test.describe.serial('checkout', () => {
  test('add to cart', ...);
  test('enter address', ...);
  test('pay', ...);
});

Worker Isolation

  • Each worker = separate process
  • Own browser instance
  • No shared state between workers
  • Fixtures scoped to worker or test

Best Practices

1. Use getByRole and getByLabel over CSS selectors
2. Never use hard-coded waits (page.waitForTimeout)
3. Keep tests isolated — no shared state between tests
4. Use Page Object Model for large test suites
5. Mock external APIs to avoid flakiness
6. Run tests in parallel with fullyParallel: true
7. Use storageState to skip login in every test
8. Enable traces on retry for CI debugging
9. Use test.describe to group related tests
10. Add data-testid for elements without good ARIA roles
